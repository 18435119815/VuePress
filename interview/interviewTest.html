//将下列数组变成一维数组，并作降序处理
let arr = [[1, 2, 2], 3, [[4, 4, 6, 7, [8]]], [11, 10, 9], 12, [13]]

//先扁平化+去重
let newArr = [];
function one(arr){
    for(let i=0;i<arr.length;i++){
        if(arr[i] instanceof Array){
            one(arr[i])
        }else{
            if(newArr.indexOf(arr[i])==-1){
                newArr.push(arr[i])
            }
        }
    }

}
one(arr);
